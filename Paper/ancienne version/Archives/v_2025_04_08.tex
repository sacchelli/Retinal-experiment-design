\section{Context}

We consider a linear, stochastic, dynamical system:
%
\begin{equation} \label{eq:SD}
d\vX\,=\,\cL.\vX \,dt \,+\,\vF(t) \,dt \,+\, \sigma_B \,d\vW(t),
\end{equation}
%
where $\vX \in \setR^N$ is the state vector, with entries $X_i$, $i=1 \dots N$, $\cL$ is a $N \times N$ matrix. We assume $\cL$ to be diagonalisable, with eigenvalues
$\lambda_n, n=1 \dots N$, left (normalised) eigenvectors $\tvw_n$ - where $T$ denotes the transpose - and right normalised eigenvectors $\vv_n$.
We note $\Lambda$ the diagonal form of $\cL$ so that $\Lambda\,=\,\cP^{-1}.\cL.\cP$ where $\cP^{-1}$ is the matrix whose rows are the
left eigenvectors $\tvw_n$ while $\cP$ is the matrix whose columns are the right eigenvectors $\vv_n$. 
We have thus (spectral decomposition theorem):
%
\begin{equation}\label{eq:SpectralDecomposition}
\cL\,=\,\sum_{n=1}^N \lambda_n \, \vv_n.\tvw_n,
\end{equation} 
%
and:
%
\begin{equation}\label{eq:Orthogonality}
\tvw_m.\vv_n\,= \, \delta_{mn},
\end{equation}
%
where $\delta_{mn}$ is the Kronecker symbol. Note that the $\vv_n$s are not orthogonal in general, neither nor the $\tvw_n$s. Moreover, eigenvalues and eigenvectors are in general complex. We assume that all $\cL$ is stable, that is, its eigenvalues have a strictly negative real part. More precisely, we order the eigenvalues such that:
%
\begin{equation}\label{eq:OrderEigenvalues}
\re{\lambda_N} \leq \re{\lambda_{N-1}} \leq \dots \leq \re{\lambda_1} < \gamma,
\end{equation}
%
for some $\gamma < 0$. As a consequence, $\cL$ is invertible.

In eq. \eqref{eq:SD}, $\vF(t)$ is a smooth time dependent vector, called from now on the "input", with entries $F_i(t)$, $i=1 \dots N$. $d\vW$ is a Brownian noise, whose entries $dW_{i}(t)$ obey $\Exp{\vW}{dW_{i}(t)}=0$,
$\Exp{\vW}{dW_{i}(t)dW_{i'}(t')}\,=\,\delta(t-t')\delta_{ii'}$ where we note
$\Exp{\vW}{}$ the expectation with respect to $\vW$.\\

We assume that the dynamical system \eqref{eq:SD} was initially prepared in a distant past, $t_0$, and that the input
$\vF$ is applied starting from $t=0$. The solution of \eqref{eq:SD} is then:
%
\begin{equation}\label{eq:SolSDGen}
\vX(t)\,=\, e^{\cL(t-t_0)}.\vX(t_0) \,+\,  \int_{0}^t e^{\cL(t-s)}.\vF(s) ds \,+\, \sigma_B \, \int_{t_0}^t e^{\cL(t-s)}.d\vW(s). 
\end{equation}
%
Using the stability of $\cL$ we let the initial time $t_0$ tend to $-\infty$ so as to remove the dependence on the initial condition, $\vX(t_0) $. The solution becomes, for $t>0$:
%
\begin{equation}\label{eq:SolSD}
\vX(t)\,=\, \int_{0}^t e^{\cL(t-s)}.\vF(s) ds \,+\, \sigma_B \, \int_{-\infty}^t e^{\cL(t-s)}.d\vW(s). 
\end{equation}
%
This is a Gaussian stochastic process with mean:
%
\begin{equation}\label{eq:Exp_X}
\mu(t) \deq \Exp{\vW}{\vX(t)} \,=\, \int_{0}^t e^{\cL(t-s)}.\vF(s) ds, 
\end{equation}
%
and covariance:
%
\begin{equation}\label{eq:Cov_X}
\cC(t,t') \deq \Exp{\vW}{\pare{\vX(t)-\mu(t)}.\pare{\vX(t')-\mu(t')}^T} \,=\, \sigma_B^2 \, \int_{-\infty}^t e^{\cL(t-s)}.e^{\cLT(t'-s)} \, ds, 
\end{equation}
%
for $t\leq t'$. Note that $\cL$ does in general  not commute with its transpose. The covariance can be expressed in terms of eigenvalues and eigenvectors using the spectral decomposition theorem \eqref{eq:SpectralDecomposition}:
%
%$$
%\cC(t,t') \,=\, \int_{-\infty}^t \pare{\sum_{n=1}^N e^{\lambda_n(t-s)} \, \vv_n.\tvw_n}.\pare{\sum_{m=1}^N e^{\lambda_m(t-s)} \, \vw_m.\tvv_m} \,ds \,=\, \sum_{n,m=1}^N \,\vv_n.\tvw_n.\vw_m.\tvv_m \int_{-\infty}^t e^{\pare{\lambda_n+\lambda_m}(t-s)} \, ds
%$$
%
\begin{equation}\label{eq:Cov_X_Spectral}
\cC(t,t') \,=\, -\, \sum_{n=1}^N \,\vv_n.\tvw_n. \sum_{m=1}^N \frac{e^{\lambda_m(t'-t)}}{\lambda_n+\lambda_m}.\vw_m.\tvv_m
\,=\, -\,  \sum_{m=1}^N e^{\lambda_m(t'-t)} \sum_{n=1}^N \, \frac{\vv_n.\tvw_n.\vw_m.\tvv_m}{\lambda_n+\lambda_m}
\end{equation}
%%
Remember that the eigenvalues have a negative real part, and, when complex, go by conjugate pairs. So, correlation is real, as expected. It depends only on the difference $t'-t$ (stationarity) because we took the limit of the initial time tending to $-\infty$. 

The derivation of the covariance is done in the appendix \ref{Sec:ACov}.
A simpler form can be obtained introducing the metric:
%
\begin{equation}\label{eq:Metric}
\cG\,=\,\cP^{-1}.\pare{\cP^{-1}}^T,
\end{equation}
%
with entries:
\begin{equation}\label{eq:gij}
g_{mn}\,=\,<\vw_m,\vw_n>,
\end{equation}
%
and the matrix:
%
\begin{equation}\label{eq:Delta}
\cD(t,t')\,=\,\cP^{-1}.\cC(t,t').\pare{\cP^{-1}}^T.
\end{equation}

Then:
%
\begin{equation}\label{eq:Deltamn}
\cD_{mn}(t,t') \,=\,  
- \, \sigma_B^2 \, \frac{g_{mn}}{\lambda_m \,+\, \lambda_n} \, e^{\lambda_n \, \pare{t'-t}}.
\end{equation}

Defining the matrix $\cA$ with entries:
%
\begin{equation}\label{eq:Amn}
A_{mn}=\frac{g_{mn}}{\lambda_m \,+\, \lambda_n},
\end{equation}
%
we have:
%
\begin{equation}\label{eq:D_mat}
\cD(t,t') \,=\,- \, \sigma_B^2 \cA.e^{\Lambda(t-t')}.
\end{equation}

It is also useful to compute the inverse of $\cD(t,t')$, $\cD^{-1}(t,t')\,=\ \cP^T.\cC^{-1}(t,t').\cP$, which is:
%
\begin{equation}\label{eq:Dmoins1}
\cD^{-1}(t,t') \,=\, - \, \sigma_B^{-2} \, e^{-\Lambda(t-t')}.\cA^{-1},
\end{equation} 
%
with entries:
%
\begin{equation}\label{eq:Dmoins1}
\cD^{-1}_{mn}(t,t') \,=\, - \, \sigma_B^{-2} \, e^{-\lambda_m(t-t')}.\cA^{-1}_{mn},
\end{equation} 

Note that this matrix diverges as $t-t' \to +\infty$, that is, the longer the time interval $t-t'$, the larger the weight of $\cD^{-1}(t,t')$.

\section{Estimation problem}

The matrix $\cL$ depends on a set of parameters, noted in the form of a vector $\bar\mtb$ of dimension $P$. The problem is to estimate at best these parameters by using a loop strategy: (1), Present a series of inputs $\vF^{(k,1)}(t), \, k=1 \dots K$, distributed according to a probability density $\xi^{(1)}$;
(2), Estimate $\bar\mtb$ by minimizing some distance between the measured response and the theoretical response \eqref{eq:SolSD}; (3), From this estimation propose a new series of input $\vF^{(k,2)}(t), \, k=1 \dots K$, optimal for some criterion, and distributed according to a probability density $\xi^{(2)}$. And so on. 
In the following, we use the methodology proposed in \cite{pronzato:08} and adapt our case to this setting.

\subsection{Input structure and notations}\label{eq:InputStructure}

Although time is continuous, observations are made on a discrete time with a step $\delta>0$, over a time horizon $[0,(H-1)\delta]$. Thus, the input $\vF$ is a piecewise constant function\footnote{We drop here the dependence in $k$ to alleviate the notations.} of time taking the value $\vF(t_l)$ on the time interval $[t_l, t_{l+1}[$, with $t_l=(l-1)\delta$. $\vF$ can therefore be represented as a list of $H$ vectors of dimension $N$, $\left\{\vF(t_0), \dots, \vF(t_{H-1})\right\}$ that we convert into a  one dimensional vector, $\vu$, of dimension $Q=N \times H$, with entries $u_q\,=\,F_i(t_l)$, with $q=i+(l-1)N$. 

We assume here, for simplicity, that $\delta$ is quite smaller than the characteristic times in the problem so that we may approximate the differential equation \eqref{eq:SD} by an Euler discretisation:
%
$$
\vX(t_l) \,-\, \vX(t_{l-1})\,=\, \delta \cL \vX(t_{l-1}) \,+\, \delta \vF(t_{l-1}) \,+\, \sigma_B \vN(t_{l}),
$$
%
where $\vN(t_{l})$ is a Gaussian vector, centered, with covariance
%
$$
\Exp{}{\vN(t_{l}).\vN^T(t_{l'})} \,=\, \delta \cI_N \delta_{l l'},
$$
%
$\cI_N$ being the $N$ dimensional identity matrix and $\delta_{ll'}$ the Kronecker symbol.
% %
Without loss of generality, because the system is linear, we can assume that the stimulation starts with the initial condition $\vX(0)=0$. In addition, like in the previous section, we assume that the noise was applied in a distant past $\delta n_0$ where $n_0 \to -\infty$. Then:
%
\begin{equation}\label{eq:SDD}
\vX(t_l)  \,=\, \delta  \sum_{n=0}^{l-1} \cM^{l-n-1} \vF(t_n) \,+\, \sigma_B \sum_{n=n_0}^{l}  \cM^{l-n} \vN(t_{n})
\end{equation}
%
where $\cM \,=\, \cI_N \,+\, \delta \cL$.
%
With a slight abuse of notation\footnote{Because $\vX(t_l)$ does not depend on the whole input vector $\vu$, only on the entries corresponding to times $< t_l$} we note $\yb(t_\ell,\ub) \,=\,\vX(t_l)$. We also note $\etab(t_\ell,\bar\mtb,\ub) \,=\, \delta  \sum_{n=0}^{l-1} \cM^{l-n-1} \vF(t_n)$ and $\mveb(t_\ell) \,=\, \sigma_B \sum_{n=n_0}^{l}  \cM^{l-n} \, \vN(t_{n})$ so that \eqref{eq:SDD} reads:
%
\begin{equation}\label{eq:SDE}
\yb(t_\ell,\ub) = \etab(t_\ell,\bar\mtb,\ub)+\mveb(t_\ell),
\end{equation}
%
where $\mveb(t_\ell)$ is called the "error".

 As a sum of Gaussian centered variables the error $\mveb(t_\ell)$ is Gaussian, centered. Its covariance $\Sigma(t_l,t_{l'}) \,=\, \Exp{}{\mveb(t_\ell)\mveb^T(t_{l'})}$ obeys:
%
$$
\Sigma(t_l,t_{l'}) \,=\, \sigma_B^2 \, \sum_{n=n_0}^{l} \sum_{n'=n_0}^{l'} \cM^{l-n} . \underbrace{\Exp{}{\vN(t_{n}).\vN^T(t_{n'})}}_{\delta \cI_N \delta_{n n'}} . \pare{\cM^T}^{l'-n'}
\,=\, \delta \,\sigma_B^2 \,  \sum_{n=n_0}^{l} \cM^{l-n} . \pare{\cM^T}^{l'-n},
$$
%
assuming $l \leq l'$. Using the diagonal form of $\cM$ this is:
%
$$
\Sigma(t_l,t_{l'}) \,=\, \delta \,\sigma_B^2 \,  \cP \pare{\sum_{n=n_0}^{l} \pare{\cI_N\,+\, \delta \Lambda}^{l-n}.\underbrace{\cP^{-1}.\cP^{{-1}^T}}_{\cG}.\pare{\cI_N\,+\, \delta \Lambda}^{l'-n}}.\cP^T
$$
%
Noting $\cD(t_l,t_{l'}) \,=\,\cP^{-1}.\Sigma(t_l,t_{l'}).{\cP^T}^{-1}$ and denoting $D_{ik}(t_l,t_{l'})$ the entry $i,k$ of the $N \times N$ matrix $\cD(t_l,t_{l'})$, we have:
%
$$
D_{ik}(t_l,t_{l'}) \,=\, \delta \,\sigma_B^2 \,  g_{ik} \, \sum_{n=n_0}^{l} \pare{1+\delta \lambda_i}^{l-n}\pare{1+\delta \lambda_k}^{l'-n}
$$
%
$$
\,=\, \delta \,\sigma_B^2 \,  g_{ik} \, \pare{1+\delta \lambda_i}^{l}\pare{1+\delta \lambda_k}^{l'} \, \sum_{n=n_0}^{l} \pare{1+\delta \lambda_i}^{-n}\pare{1+\delta \lambda_k}^{-n}.
$$
%
We set $q=\pare{1+\delta \lambda_i}\pare{1+\delta \lambda_k} = 1 + \delta \pare{\lambda_i+\lambda_k} + \delta^2 \lambda_i\lambda_k$. We assume that $\delta$ is small enough so that $q \sim 1 + \delta \pare{\lambda_i+\lambda_k}$ and that $|q|<1$ (which holds for sufficiently small $\delta$ as all eigenvalues are stable\footnote{
$\abs{1+\delta\lambda}^2 = \abs{1+\delta \lambda_r + i \delta \lambda_i}^2$ 
$=\pare{1+\delta \lambda_r}^2 + \delta^2 \lambda_i^2  = 1 +2 \delta \lambda_r + \delta^2\pare{\lambda_r^2+\lambda_i^2}  < 1$  
$\Leftarrow \delta < - \frac{2 \lambda_r}{\lambda_r^2+\lambda_i^2}$ }
).
%
Then,
%
$$\sum_{n=n_0}^{l} \pare{1+\delta \lambda_i}^{-n}\pare{1+\delta \lambda_k}^{-n}
=\sum_{n=n_0}^{l} q^{-n} 
=q^{-n_0} \frac{1-q^{n_0-l+1}}{1-q}
=\frac{q^{-n_0}-q^{-l+1}}{1-q} 
 \to
- \frac{q^{-l+1}}{1-q}
,$$
%
as $n_0 \to -\infty$.\\
%
It follows that:
%
$$
D_{ik}(t_l,t_{l'}) \,=\, \delta \,\sigma_B^2 \,  g_{ik} \,
\pare{1+\delta \lambda_i}^{l}\pare{1+\delta \lambda_k}^{l'} \, \frac{\pare{1+\delta \lambda_i}^{-l+1}\pare{1+\delta \lambda_k}^{-l+1}}{\delta \pare{\lambda_i+\lambda_k} + \delta^2 \lambda_i\lambda_k}
$$
%
$$
\,=\, \sigma_B^2 \,  g_{ik} \, \frac{\pare{1+\delta \lambda_i}\pare{1+\delta \lambda_k}}{\pare{\lambda_i+\lambda_k} + \delta \lambda_i\lambda_k} \, \pare{1+\delta \lambda_k}^{l'-l},
$$
%
which is similar to \eqref{eq:Deltamn}.
%
% $$
% \cD_{mn}(t,t') \,=\,  
% - \, \sigma_B^2 \, \frac{g_{mn}}{\lambda_m \,+\, \lambda_n} \, e^{\lambda_n \, \pare{t'-t}}.
% $$
%
Note that the covariance is therefore not diagonal, although its off diagonal terms decay exponentially fast.

We make observations $\yb(t_\ell,\ub)$ at times $t_\ell=(\ell-1)\delta$, $\ell=1,\ldots,H$. Each $\yb(t_\ell,\ub)$ is a $N$-dimensional vector (all vectors are column vectors) and corresponds to the model response to the piecewise constant input $\ub$ (an $H$-dimensional vector) corrupted by noise; that is,
\beq\label{model1}
\yb(t_\ell,\ub) = \etab(t_\ell,\bar\mtb,\ub)+\mveb(t_\ell) \,,
\eeq
with $\bar\mtb$ an unknown vector of $p$ parameters. We assume that the dynamic of the system is fast enough to make the errors $\mveb(t_\ell)$ and $\mveb(t_{\ell'})$ independent for $t_\ell\neq t_{\ell'}$ and denote $\Sigmab_\ell=\Ex\{\mveb(t_\ell)\mveb\TT(t_\ell)\}$, an $N \times N$ matrix, not necessarily diagonal.

If the observations at different time instants were uncorrelated, with $\Sigmab_\ell$ known (not dependent of $\mtb$) for each $\ell$, 
the (weighted) LS criterion for the input $\ub$ would be: 
\bea
J(\mtb,\ub) = \sum_{\ell=1}^H \left[\yb(t_\ell,\ub) - \etab(t_\ell,\mtb,\ub)\right]\TT\Sigmab_\ell^{-1}\left[\yb(t_\ell,\ub) - \etab(t_\ell,\mtb,\ub)\right] \,.
\eea
When using all observations available for different inputs $\ub^{(k)}$, $k=1,\ldots,K$, we thus get the following criterion (after normalisation by $1/K$):
\bea
J_K(\mtb) = \frac1K\, \sum_{k=1}^K \sum_{\ell=1}^H \left[\yb(t_\ell,\ub^{(k)}) - \etab(t_\ell,\mtb,\ub^{(k)})\right]\TT\Sigmab_\ell^{-1}\left[\yb(t_\ell,\ub^{(k)}) - \etab(t_\ell,\mtb,\ub^{(k)})\right] \,.
\eea

In the following we account for the correlation between observations at different time instants (but assume that observations for different inputs $\ub^{(k)}$ correspond to different experiments and are uncorrelated). 
Denoting $\Yb(\ub)=\left[\yb\TT(t_1,\ub),\ldots,\yb\TT(t_H,\ub)\right]\TT$ and $\zetab(\mtb,\ub)=\left[\etab\TT(t_1,\mtb,\ub),\ldots,\etab\TT(t_H,\mtb,\ub)\right]\TT$, we get
\beq\label{LS-Sigma-known}
J_K(\mtb) = \frac1K\, \sum_{k=1}^K \left[\Yb(\ub^{(k)}) - \zetab(\mtb,\ub^{(k)})\right]\TT \Sigmab^{-1}\left[\Yb(\ub^{(k)}) - \zetab(\mtb,\ub^{(k)})\right] \,,
\eeq

\subsection{Design of experiments: $\Sigmab$ is known (does not depend on $\bar\mtb$)}\label{S:Sigma-known}

Here we assume that $\Sigmab$ is known: it does not depend on $\bar\mtb$.

We consider the situation when $H$ is fixed but $K$ tends to infinity. Let $\xi$ denote a probability measure on the set $\SU\subset \mathds{R}^H$ of input vectors $\ub$. We assume that $\SU$ is compact and that
\begin{itemize}
  \item either $\xi$ has finite support and the empirical measure $\xi_{e,K}$ of the $\ub^{(k)}$ converges strongly to $\xi$ as $K\to\infty$ (i.e., $\lim_{K\to\infty}\xi_{e,K}(\{\ub\})=\xi(\{\ub\})$ for all $\ub\in\SU$),

  \item or the $\ub^{(k)}$ are independently sampled from $\xi$.
\end{itemize}

Then, under classical regularity assumptions on the functions $\etab(t,\cdot,\ub)$ and on the admissible set $\Theta$ for $\mtb$ ($\Theta$ is compact and $\bar\mtb\in \mathrm{int}(\Theta)$), the LS estimator $\hat\mtb^K$ that minimises $J_K(\mtb)$ with respect to $\mtb\in\Theta$ converges a.s.\ to $\bar\mtb$ when $K\to\infty$ and is asymptotically normal
\bea
\sqrt{K} (\hat\mtb^K-\bar\mtb) \rad \zb \sim \SN\left(\0b,\Mb^{-1}(\xi,\bar\mtb)\right)\,, \ K\to\infty\,,
\eea
where $\Mb(\xi,\mtb)$ denotes the information matrix for model \eqref{model1}:
\bea
\Mb(\xi,\mtb) = \int_\SU \frac{\mp\zetab\TT(\mt,\ub)}{\mp\mtb} \Sigmab^{-1} \frac{\mp\zetab(\mt,\ub)}{\mp\mtb\TT}\, \xi(\dd\ub) \,,
\eea
see, e.g., \cite[Chap.~3]{PP2013}, \cite[Chap.~1]{FedorovL2014}.

Optimal design (approximate theory) aims at determining an optimal probability measure $\xi^*$ on $\SU$ that minimises a suitable scalar function $\Phi(\cdot)$ of $\Mb(\xi,\bar\mtb)$. As $\bar\mtb$ is unknown, we consider locally optimal design which relies on $\Mb(\xi,\mtb^0)$ with $\mtb^0$ a nominal value for $\mtb$. Since each matrix
\bea
\Mb(\delta_\ub,\mtb)= [\mp\zetab\TT(\mt,\ub)/\mp\mtb] \Sigmab^{-1} [\mp\zetab(\mt,\ub)/\mp\mtb\TT]
\eea
has rank at least one, Caratheodory's theorem implies that for any $\xi$ on $\SU$ we can find a discrete $\xi'$ on $\SU$ supported on $m\leq p(p+1)/2+1$ points such that $\Mb(\xi',\mtb)=\Mb(\xi,\mtb)$. When $\Phi(\cdot)$ is monotone, $\Mb(\xi^*,\mtb)$ is on the boundary of the set $\{\Mb(\delta_\ub,\mtb):\ \ub\in\SU\}$ and $m\leq p(p+1)/2$ support points are enough. When $\rank[\Mb(\delta_\ub,\mtb)]=\dim(\mtb)$, the optimal design may correpond to a delta measure $\delta_\ub$ at a single input $\ub\in\SU$.

A typical design criterion corresponds to $D$-optimality, with $\Phi(\Mb)=-\log\det \Mb$. A multivariate extension, see for instance \cite{HarmanT2009}, of Kiefer-Wolfowitz (1960)  equivalence Theorem \cite{KieferW60} then states that an optimal $\xi^*$ for $\mtb^0$, a nominal value for the unknown true parameters  $\bar\mtb$, minimises $\max_{\ub\in\SU} d(\ub,\xi,\mtb^0)$, with, moreover, $\max_{\ub\in\SU} d(\ub,\xi^*,\mtb^0)=p$, where
\beq\label{D-function-Sigma-known}
d(\ub,\xi,\mtb) = \tr\left[ \frac{\mp\zetab\TT(\mt,\ub)}{\mp\mtb} \Sigmab^{-1} \frac{\mp\zetab(\mt,\ub)}{\mp\mtb\TT} \Mb^{-1}(\xi,\mtb) \right] \,.
\eeq

The vertex-direction algorithm of Fedorov (1972) \nocite{Fedorov72} %\Luccomment{Celui du \S 2.3.2 de la note de Bruno}
corresponds to a Frank-Wolfe algorithm for the maximization of $\log\det \Mb(\xi,\mtb^0)$. Starting from a measure $\xi^{(0)}$ such that $\Mb(\xi^{(0)},\mtb^0)$ is nonsingular, 
%\Luccomment{La non singularit\'e de $\Mb$ est essentielle.} 
at iteration $k\geq 0$ the algorithm chooses $\ub^{(k+1)}\in\Arg\max_{\ub\in\SU} d(\ub,\xi^{(k)},\mtb^0)$ and then sets
\bea
\xi^{(k+1)}=(1-\ma)\xi^{(k)}+\ma\delta_{\ub^{(k+1)}} \,.
\eea
The optimal value $\ma^*$ of the step size $\ma$ is known explicitly in the single response case, but must be determined numerically \B{How ?}\Luc{C'est un problĂ¨me convexe de dimension 1, une mĂ©thode de type section dorĂ©e ou Fibonacci suffit} in the multi-response situation considered here.

An alternative construction of an optimal design, which does not require differentiation of $\zetab(\mt,\ub)$ with respect to $\mtb$ (and thus no calculation of the information matrix) and does not rely on asymptotic considerations is proposed in \cite[Chap.~7]{PP2013} and \cite{PazmanP2014}. The criterion (to be maximised) is
\beq\label{phi-Sigma-known}
\phi(\xi) = \min_{\mtb\in\Theta} \frac{\int_\SU \left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]\TT \Sigmab^{-1}\left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]\,\xi(\dd\ub)}{\|\mtb-\mtb^0\|^2}\,,
\eeq
where $\Theta$ is a compact subset of $\mathds{R}^p$ containing admissible parameter vectors for the model. If we discretise $\Theta$ into a finite set $\Theta_Q=\{\mtb^{(1)},\ldots,\mtb^{(Q)}\}$ and consider discrete measures $\xi$ that set weights $w_i$ to $L$ support points $\ub^{(1)},\ldots,\ub^{(L)}$, the criterion becomes
\bea
\phi(\wb) = \min_{\mtb^{(j)}\in\Theta_Q} \frac{\sum_{i=1}^L w_i\, \left[\zetab(\mtb^{(j)},\ub^{(i)}) - \zetab(\mtb^0,\ub^{(i)})\right]\TT \Sigmab^{-1}\left[\zetab(\mtb^{(j)},\ub^{(i)}) - \zetab(\mtb^0,\ub^{(i)})\right]}{\|\mtb-\mtb^0\|^2}\,.
\eea
Its maximisation with respect to $\wb=(w_1,\ldots,w_L)\TT$ subject to $\sum_{i=1}^L w_i = 1$ corresponds to an LP problem:
\bea
&& \mbox{Maximise $t$ with respect to $t$ and $\wb\in\mathds{R}^L$} \B{N'est ce pas "Maximize $\phi$ ... ? }\\
%
&&\Luc{La formulation du pb de programmation lineaire dans $R^{L+1}$ me semble correcte}\\

&& \hspace{1cm} \sum_{i=1}^L w_i \frac{\left[\zetab(\mtb^{(j)},\ub^{(i)}) - \zetab(\mtb^0,\ub^{(i)})\right]\TT \Sigmab^{-1}\left[\zetab(\mtb^{(j)},\ub^{(i)}) - \zetab(\mtb^0,\ub^{(i)})\right]}{\|\mtb-\mtb^0\|^2} \geq t \,, \\ && \hspace{9cm} j=1,\ldots,Q \,.

\subsection{Design of experiments: $\Sigmab=\Sigmab(\mtb)$ depends on $\mtb$}

\B{This is the case with our problem.}
A first possibility in that case is to use iteratively reweighted LS:
\begin{enumerate}
    \item set an initial value for $\Sigmab=\Sigmab^0$, for instance $\Sigmab(\mtb^0)$ with $\mtb^0$ a nominal value for $\mtb$;
    \item estimate $\mtb$ by minimising \eqref{LS-Sigma-known} with $\Sigmab=\Sigmab^0$; this gives $\hat\mtb^K_0$;
    \item repeat the estimation of $\mtb$ with now $\Sigmab=\Sigmab(\hat\mtb^K_0)$ in \eqref{LS-Sigma-known}; this gives $\hat\mtb^K_1$; substitute $\hat\mtb^K_1$ for $\hat\mtb^K_0$ and repeat, etc.
\end{enumerate}
From the asymptotic point of view (as $K\to\infty$), stopping at $\hat\mtb^K_1$ (i.e., two-stage LS) is enough, but this iteratively reweighted scheme can be continued until stage $m$ where $\hat\mtb^K_m \simeq \hat\mtb^K_{m-1}$.

Another possibility, which we develop below, is to use penalized LS: we assume that $\Yb(\ub)$ is normally distributed $\SN(\zetab(\bar\mtb,\ub),\Sigmab(\bar\mtb))$ \B{which is OK} and estimate $\mtb$ by maximum likelihood. The criterion to be minimised with respect to $\mtb$ is then
\beq
J_K(\mtb) &=& \frac1K\, \sum_{k=1}^K \left[\Yb(\ub^{(k)}) - \zetab(\mtb,\ub^{(k)})\right]\TT \Sigmab^{-1}(\mtb)\left[\Yb(\ub^{(k)}) - \zetab(\mtb,\ub^{(k)})\right] \nonumber \\
&& + \frac1K\,  \log\det \Sigmab(\mtb)\,, \label{penalised-LS}
\eeq
that is, due to the presence of the parameterised covariance $\Sigmab(\mtb)$, the penalty term $(1/K)\,  \log\det \Sigmab(\mtb)$ is added to the LS criterion \eqref{LS-Sigma-known}.
The information matrix is now 
\bea
\Mb(\xi,\mtb) = \int_\SU \Ex\left\{ \frac{\mp\log \varphi[\Yb(\ub)|\mtb]}{\mp\mtb} \frac{\mp\log \varphi[\Yb(\ub)|\mtb]}{\mp\mtb\TT}\right\} \, \xi(\dd\ub) \,,
\eea
where $\varphi[\Yb(\ub)|\mtb]$ denotes the p.d.f.\ of the normal distribution $\SN(\zetab(\mtb,\ub),\Sigmab(\mtb))$ and the expectation $\Ex\{\cdot\}$ is calculated for this distribution. Direct calculation gives, for all $i,j=1,\ldots,\dim(\mtb)$ % \B{NOt clear how you arrive to this, but OK},
\bea
\{\Mb(\xi,\mtb)\}_{i,j} &=& \int_\SU \frac{\mp\zetab\TT(\mt,\ub)}{\mp\mtb_i} \Sigmab^{-1}(\mtb) \frac{\mp\zetab(\mt,\ub)}{\mp\mtb_j}\, \xi(\dd\ub) \\
&& + \frac12\, \int_\SU \tr\left[\Sigmab^{-1}(\mtb)\frac{\mp\Sigmab(\mtb)}{\mp\mtb_i}\Sigmab^{-1}(\mtb)\frac{\mp\Sigmab(\mtb)}{\mp\mtb_j}\right]\,\xi(\dd\ub) \,.
\eea
Since $\Sigmab(\mtb)$ does not depend on $\ub$, we get 
\bea
\Mb(\xi,\mtb) &=& \int_\SU \frac{\mp\zetab\TT(\mt,\ub)}{\mp\mtb} \Sigmab^{-1}(\mtb) \frac{\mp\zetab(\mt,\ub)}{\mp\mtb\TT}\, \xi(\dd\ub) +  \Qb(\mtb) \,,
\eea
where 
\bea
\{\Qb\}_{i,j}=\frac12\, \tr\left[\Sigmab^{-1}(\mtb)\frac{\mp\Sigmab(\mtb)}{\mp\mtb_i}\Sigmab^{-1}(\mtb)\frac{\mp\Sigmab(\mtb)}{\mp\mtb_j}\right] \,.
\eea

The function $d(\cdot,\xi,\mtb)$ given by \eqref{D-function-Sigma-known}, which is used in a vertex-direction algorithm and to check the $D$-optimality of a given design, see Section~\ref{S:Sigma-known}, must be modified accordingly into
\bea
d(\ub,\xi,\mtb) = \tr\left\{\left[ \frac{\mp\zetab\TT(\mt,\ub)}{\mp\mtb} \Sigmab^{-1} \frac{\mp\zetab(\mt,\ub)}{\mp\mtb\TT}+ \Qb(\mtb)\right] \Mb^{-1}(\xi,\mtb) \right\} \,.
\eea

If $\bar\mtb$ denotes the true (unknown) value of $\mtb$, then, as $K\to\infty$, under the same assumptions about the regularity of the model and the distribution of the $\ub^{(k)}$ as in Section~\ref{S:Sigma-known}, $J_K(\mtb)$ tends to 
\bea
J_{\bar\mtb}(\mtb) &=& \int_\SU \left[\zetab(\mtb,\ub^{(k)}) - \zetab(\bar\mtb,\ub^{(k)})\right]\TT \Sigmab^{-1}(\mtb)\left[\zetab(\mtb,\ub^{(k)}) - \zetab(\bar\mtb,\ub^{(k)})\right] \, \xi(\dd\ub) \\
&& + \int_\SU \left\{\tr\left[\Sigmab(\bar\mtb)\Sigmab^{-1}(\mtb)\right] - \log \det\left[\Sigmab(\bar\mtb)\Sigmab^{-1}(\mtb)\right] \right\} \, \xi(\dd\ub) \\
&& + \int_\SU \log \det \Sigmab(\bar\mtb)\, \xi(\dd\ub) \\
&=& \int_\SU \left[\zetab(\mtb,\ub^{(k)}) - \zetab(\bar\mtb,\ub^{(k)})\right]\TT \Sigmab^{-1}(\mtb)\left[\zetab(\mtb,\ub^{(k)}) - \zetab(\bar\mtb,\ub^{(k)})\right] \, \xi(\dd\ub) \\

\eea
and the convergence is uniform in $\mtb$. 
The criterion $J_{\bar\mtb}(\cdot)$ is minimum at $\mtb=\bar\mtb$, with $\mp J_{\bar\mtb}(\mtb)/\mp\mtb_{|\mtb=\bar\mtb}=\0b$.

Instead of maximising $\phi(\xi)$ given by \eqref{phi-Sigma-known}, we must now maximise
\beq\label{phi-penalised-LS}
\phi(\xi)= \min_{\mtb\in\Theta}  \frac{J_{\mtb^0}(\mtb)-J_{\mtb^0}(\mtb^0)}{\|\mtb-\mtb^0\|^2} = \min_{\mtb\in\Theta}  \frac{\int_\SU KL(\mtb||\mtb^0;\ub)\,\xi(\dd\ub)}{\|\mtb-\mtb^0\|^2} \,,
\eeq
where $KL(\mtb||\mtb^0;\ub)$
denotes the (nonsymmetric) Kullback-Leibler divergence between the normal distributions for models with parameters $\mtb$ and $\mtb^0$, with $\mtb^0$ a nominal value for the unknown true parameters  $\bar\mtb$:
\bea
KL(\mtb||\mtb^0;\ub) &=&\int \log\left\{\frac{\varphi[\Yb(\ub)|\mtb]}{\varphi[\Yb(\ub)|\mtb^0]}\right\}\, \varphi[\Yb(\ub)|\mtb]\,\dd\Yb \\
&=& \left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]\TT \Sigmab^{-1}(\mtb)\left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]  \\
&& +\, \tr\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right] - \log \det\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right] - \dim(\zetab)\,,
\eea
with $\dim(\zetab)=NH$. 

\B{Thus:
%
$$
\phi(\xi)\,=\, 
\min_{\mtb\in\Theta} \frac{1}{\|\mtb-\mtb^0\|^2} \,  \pare{ \begin{array}{ll}
&\int_\SU \left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]\TT \Sigmab^{-1}(\mtb)\left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right] \, \xi(\dd\ub)\\
&+\, \tr\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right] - \log \det\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right] - \dim(\zetab).
\end{array}
%
}
$$
%
Denoting $\vs_i(\mtb)$ the normalized eigenvectors of $\Sigmab(\mtb)$, which are real and orthogonal, and $\sigma_i(\mtb)$ the corresponding real non negative eigenvalues
we can project $\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)$ on this basis, $\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub) = \sum_{i=1}^{NH} \pi_i(\mtb,\ub) \vs_i(\mtb)$ and write:
%
$$
\int_\SU \left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right]\TT \Sigmab^{-1}(\mtb)\left[\zetab(\mtb,\ub) - \zetab(\mtb^0,\ub)\right] \, \xi(\dd\ub)
=\sum_{i=1}^{NH}  \frac{1}{\sigma_i(\mtb)} \int_\SU \pi_i^2(\mtb,\ub) \, \xi(\dd\ub).
$$
%
Note that this is the only term in the criterion $\phi$ which depends on $\xi$.
Also, we have:
%
$$
\log \det\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right]  \,=\, \sum_{i=1}^{NH} \log \frac{\sigma_i(\mtb^0)}{\sigma_i(\mtb)}.
$$
%
Finally, since
$\Sigmab(\mtb^0) \,=\,\sum_{i=1}^{NH} \sigma_i(\mtb^0) \vs_i(\mtb^0) \vs_i(\mtb^0)^T$, $\Sigmab^{-1}(\mtb)\,=\,\sum_{j=1}^{NH} \sigma_j^{-1}(\mtb) \vs_j(\mtb) \vs_j(\mtb)^T$ we have:
%
$$
\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb) \,=\, \sum_{i,j=1}^{NH} \frac{\sigma_i(\mtb^0)}{\sigma_j(\mtb)} \, \vs_i(\mtb^0) \, \underbrace{\vs_i(\mtb^0)^T \vs_j(\mtb)}_{<\vs_i(\mtb^0),\vs_j(\mtb)>} \vs_j(\mtb)^T
$$
%
so that:
%
$$
\pare{\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)}_{kl} \,=\, <\vs_k(\mtb),\pare{\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)}\vs_l(\mtb)>
$$
%
$$
=\, \sum_{i,j=1}^{NH} \frac{\sigma_i(\mtb^0)}{\sigma_j(\mtb)} \,<\vs_k(\mtb^0),\vs_i(\mtb)> \,<\vs_i(\mtb^0),\vs_j(\mtb)>  \,  \underbrace{<\vs_j(\mtb), \vs_l(\mtb)>}_{\delta_{jl}}
$$
%
$$
=\, \sum_{i=1}^{NH} \frac{\sigma_i(\mtb^0)}{\sigma_l(\mtb)} \, <\vs_k(\mtb^0),\vs_i(\mtb)> \,<\vs_i(\mtb^0),\vs_l(\mtb)> 
$$
and:
%
$$
\tr\left[\Sigmab(\mtb^0)\Sigmab^{-1}(\mtb)\right] \,=\, 
\sum_{i,k=1}^{NH} \frac{\sigma_i(\mtb^0)}{\sigma_k(\mtb)} \, <\vs_k(\mtb^0),\vs_i(\mtb)> \,<\vs_i(\mtb^0),\vs_k(\mtb)> 
$$
}

\bigskip

\subsection{Computation of the covariance}\label{Sec:ACov}

The covariance of  $\vX(t)$, the solution of \eqref{eq:SD} is defined by
%
$$
\cC(t,t') \deq \Exp{\vW}{\pare{\vX(t)-\mu(t)}.\pare{\vX(t')-\mu(t')}^T}.
$$
%
Using the general solution \eqref{eq:SolSD} and the definition of the expectation \eqref{eq:Exp_X} this is:
%
$$
\sigma_B^2 \, \int_{-\infty}^t e^{\cL(t-s)}.\underbrace{\Exp{\vW}{d\vW(s).d\vW^T(s')}}_{\delta(s-s')\,\cI}e^{\cLT(t'-s')} \,ds\,ds'
$$
%
$$
=\,\sigma_B^2 \, \int_{-\infty}^{t \wedge t'} e^{\cL(t-s)}.e^{\cLT(t'-s)} \,ds.
$$
%
Assuming $t\leq t'$ this gives:
%
$$
\cC(t,t')  \,=\, \sigma_B^2 \, \int_{-\infty}^t e^{\cL(t-s)}.e^{\cLT(t'-s)} \, ds, 
$$
%
corresponding to \eqref{eq:Cov_X} in the main text.
Using the diagonal form of $\cL$ gives:
%
$$
\cC(t,t')  \,=\, \sigma_B^2 \, \int_{-\infty}^t \cP.e^{\Lambda(t-s)}.\cP^{-1}. \pare{\cP.e^{\Lambda(t'-s)}.\cP^{-1}}^T\, ds
$$
%
$$
=\, \sigma_B^2 \, \int_{-\infty}^t  \cP.e^{\Lambda(t-s)}.\underbrace{\cP^{-1}.\pare{\cP^{-1}}^T}_{\cG}.e^{\Lambda(t'-s)}.\cP^T.
$$

The matrix $\cG\,=\,\cP^{-1}.\pare{\cP^{-1}}^T$ can also be written in terms of the left eigenvectors of $\cL$ (the rows of $\cP^{-1}$):
%
$$
\cG \,=\, \pare{\vw^T_1, \dots, \vw^T_N}.\vect{\vw_1\\\dots\\\vw_N}.
$$
%
This is a positive, definite, matrix with entries
%
$g_{mn}\,=\,\vw^T_m.\vw_n\,=\,<\vw_m,\vw_n>$.
%
This defines therefore a metric. 

We note that $\cD(t,t') \equiv \cP^{-1}.\cC(t,t').\pare{\cP^{-1}}^T$
has entries:
%
$$
\cD_{mn}(t,t') \,=\, \sigma_B^2 \,g_{mn} \,  \int_{-\infty}^t e^{\lambda_m(t-s)}\,e^{\lambda_n(t'-s)}\, ds
\,=\, \sigma_B^2 \,g_{mn} \, e^{\lambda_m \, t \,+\, \lambda_n \, t'} \int_{-\infty}^t 
e^{-\,\pare{\lambda_m \,+\, \lambda_n}s} \, ds
$$
$$
=\, - \, \sigma_B^2 \, \frac{g_{mn}}{\lambda_m \,+\, \lambda_n} \, e^{\lambda_m \, t \,+\, \lambda_n \, t'} \, 
\bra{e^{-\,\pare{\lambda_m \,+\, \lambda_n}s}}_{-\infty}^t
$$ 
%
$$
\cD_{mn}(t,t') \,=\,  
- \, \sigma_B^2 \, \frac{g_{mn}}{\lambda_m \,+\, \lambda_n} \, e^{\lambda_n \, \pare{t'-t}},
$$
using that all eigenvalues are stable. Defining the matrix $\cA$ with entries $A_{mn}=\frac{g_{mn}}{\lambda_m \,+\, \lambda_n}$ we have
$\cD(t,t') \,=\,- \, \sigma_B^2 \cA.e^{\Lambda(t-t')}$.

It is also useful to compute the inverse of $\cD(t,t')$, $\cD^{-1}(t,t')\,=\ \cP^T.\cC^{-1}(t,t').\cP$, which is:
%
\begin{equation}\label{eq:Dmoins1}
\cD^{-1}(t,t') \,=\, - \, \sigma_B^{-2} \, e^{-\Lambda(t-t')}.\cA^{-1}.
\end{equation} 

\subsection{Computation of $\frac{\partial \cI^{l,r}}{\partial \theta_p}$}\label{Sec:dIdtheta}

From \eqref{eq:Ilr}, 
$\cI^{l,r}\,=\,\int_{\delta r}^{\delta (r+1)} e^{\cL\pare{l \delta - s}} \,ds$
%
so, $\frac{\partial \cI^{l,r}}{\partial \theta_p}\,=\, \int_{\delta r}^{\delta (r+1)} \frac{\partial e^{\cL\pare{l \delta - s}}}{\partial \theta_p} \,ds$. In general, $\frac{\partial  \cL}{\partial \theta_p}$ does not commute with $\cL$  so one cannot write $\frac{\partial e^{\cL\pare{l \delta - s}}}{\partial \theta_p}\,=\,\pare{l \delta - s} \frac{\partial \cL}{\partial \theta_p}.e^{\cL\pare{l \delta - s}}$. Instead, using the series definition of $e^{\cL\pare{t - s}}$ one can show that:
%
$$
\frac{\partial e^{\cL\pare{t - s}}}{\partial \theta_p} \,=\, \sum_{n=1}^{+\infty} \frac{\pare{t - s}^n}{n!} \pare{\sum_{m=0}^{n-1} \cL^m.\frac{\partial  \cL}{\partial \theta_p}.\cL^{n-(m+1)}}.
$$
%
Thus:
%
$$
\cP^{-1}.\frac{\partial e^{\cL\pare{t - s}}}{\partial \theta_p}.\cP\,=\, \sum_{n=1}^{+\infty} \frac{\pare{t - s}^n}{n!} \pare{\sum_{m=0}^{n-1} \Lambda^m.\cD_p.\Lambda^{n-(m+1)}}
$$
%
with:
%
\begin{equation}\label{eq:Dp}
\cD_p\,=\,\cP^{-1}.\frac{\partial  \cL}{\partial \theta_p}.\cP.
\end{equation}

We have therefore:
%
$$
\cP^{-1}.\frac{\partial \cI^{l,r}}{\partial \theta_p}.\cP \,=\, \sum_{n=1}^{+\infty} \frac{1}{n!} \pare{\sum_{m=0}^{n-1} \Lambda^m.\cD_p.\Lambda^{n-(m+1)}} \int_{\delta r}^{\delta (r+1)} \pare{l \delta - s}^n\,ds 
$$
%
$$
=\,\sum_{n=1}^{+\infty} \frac{\delta^{n+1}}{(n+1)!} \pare{\sum_{m=0}^{n-1} \Lambda^m.\cD_p.\Lambda^{n-(m+1)}} \,A_{l,r,n+1},
$$
%
with $A_{l,r,n+1} \,=\,\bra{(l-r)^{n+1} \,-\,(l-r-1)^{n+1}} \,=\, \sum_{a=1}^{n+1} \binom{n+1}{a} (-1)^{1+a} \, (l-r)^{n+1-a}.$
%
Setting $\cG^{l,r}_p=\sum_{n=1}^{+\infty} \frac{\delta^{n+1}}{(n+1)!} \pare{\sum_{m=0}^{n-1} \Lambda^m.\cD_p.\Lambda^{n-(m+1)}} \,A_{l,r,n+1}$
This gives:
%
\begin{equation}\label{eq:dIdtheta}
\frac{\partial \cI^{l,r}}{\partial \theta_p} \,=\,
\cP.\cG^{l,r}_p.\cP^{-1}.
\end{equation}

Note that we can rewrite $\cG^{l,r}_p$ in the form:
%
$$
\cG^{l,r}_p\,=\,\delta^{2} \,\sum_{n=1}^{+\infty} \frac{1}{(n+1)!} \pare{\sum_{m=0}^{n-1} \pare{\delta\Lambda}^m.\cD_p.\pare{\delta\Lambda}^{n-(m+1)}} \,A_{l,r,n+1},
$$
%
letting appear the dimensionless matrix $\delta\Lambda$ controlling the convergence rate of the series.

\subsection{Computation of the Fisher matrix in terms of the eigen-elements of $\cL$}\label{Sec:ComputeFisher}

Using \eqref{eq:DiscX}-\eqref{eq:eta} we have:
%
$$
\eta_q(\vth,\vu)\,=\, \sum_{j=1}^N \, \sum_{r=0}^{l-1} \, \cI^{l,r}_{ij}\,F_j(r\delta), \, q\,=\, i+lN.
$$
%
As $F_j(r\delta)$ does not depend on $\vth$ we have:
%
$$
\frac{\partial \eta_q(\vth,\vu)}{\partial \theta_p}
\equiv E_{qp}
\,=\,\sum_{j=1}^N \, \sum_{r=0}^{l-1} \, \frac{\partial \cI^{l,r}_{ij}}{\partial \theta_p}\,F_j(r\delta),
$$
%
corresponding to a $Q \times P$ matrix. Then:
%
$$
\bra{\pare{\frac{\partial \veta(\vth,\vu)}{\partial \vth}}^T.\pare{\frac{\partial \veta(\vth,\vu)}{\partial \vth}}}_{pp'}
\,=\, \sum_{q=1}^Q E_{pq}\,E_{qp'}
$$
$$=\, \sum_{i=1}^N \sum_{l=1}^H E_{p,i+lN}\,E_{i+lN,p'}
\,=\, \sum_{i=1}^N \sum_{l=1}^H \pare{\sum_{j=1}^N \, \sum_{r=0}^{l-1} \, \frac{\partial \cI^{l,r}_{ij}}{\partial \theta_p}\,F_j(r\delta)}
\pare{\sum_{j'=1}^N \, \sum_{r'=0}^{l-1} \, \frac{\partial \cI^{l,r'}_{ij'}}{\partial \theta_{p'}}\,F_{j'}(r'\delta)},
$$
%
$$
=\,\ \sum_{l=1}^H \sum_{r,r'=0}^{l-1} \, \sum_{i=1}^N \,\sum_{j,j'=1}^N \,\frac{\partial \cI^{l,r}_{ij}}{\partial \theta_p}\,F_j(r\delta)\,F_{j'}(r'\delta)\, \frac{\partial \cI^{l,r'}_{ij'}}{\partial \theta_{p'}}. 
$$

Therefore $\cM_F(\xi,\vth)$ as entries:
%
%
This is eq. \eqref{eq:MF_exp} in the main text, with
%
$$
\Delta_{jj'}(r,r') \,=\, \Exp{\xi}{F_j(r\delta)\,F_{j'}(r'\delta) }.
$$


Note that \eqref{eq:MF_exp} can also be written as:
%
\begin{equation}\label{eq:Fisher_Trace}
\cM_{pp'}(\xi,\vth)\,=\,  \sum_{l=1}^H \sum_{r,r'=0}^{l-1} \, Tr\bra{\frac{\partial \cI^{l,r}}{\partial \theta_p}.\Delta(r,r').\frac{\partial \cI^{l,r'}}{\partial \theta_{p'}}},
\end{equation}
%
where $l,r,r'$ are time indices and where $\Delta(r,r')\,=\,\Exp{\xi}{\vF(r\delta).\vF^T(r'\delta)}$. 

It is useful to write this matrix in terms of $\vf(r\delta)\,=\,\cP^{-1}.\vF(r\delta)$, the projection of the input in the eigenspace of $\cL$.
Using \eqref{eq:dIdtheta} we have:
%
$$
\frac{\partial \cI^{l,r}}{\partial \theta_p}.\Delta(r,r').\frac{\partial \cI^{l,r'}}{\partial \theta_{p'}}
\,=\, \cP.\cG^{l,r}_p.\cP^{-1}.\Exp{\xi}{\vF(r\delta).\vF^T(r'\delta)}.\cP.\cG^{l,r'}_p.\cP^{-1}
$$
%
$$
=\, \cP.\cG^{l,r}_p.\underbrace{\Exp{\xi}{\vf(r\delta).\vf^T(r'\delta)}}_  {\cC_f(r,r')}.\cP^T.\cP.\cG^{l,r'}_p.\cP^{-1},
$$

Thus:
%
\begin{equation}\label{eq:Fisher_Propre}
\cM_{pp'}(\xi,\vth)\,=\,  \sum_{l=1}^H \sum_{r,r'=0}^{l-1} \, Tr\bra{
\cP.\cG^{l,r}_p.\cC_f(r,r').\cP^T.\cP.\cG^{l,r'}_p.\cP^{-1}
}.
\end{equation}

\B{Que se passe-t'il si l'input est projetĂ© sur un seul vecteur propre ?}

In this decomposition, the inputs are  projected onto a fixed basis, only the coefficients of the projection are random, governed by $\xi$.
They impact the Fisher matrix via the correlation matrix $\cC_f(r,r')$ (it's a correlation if the coefficients are centered).

\subsection{Continuous-discrete model (Ludovic)}

Let us denote:
%
\begin{equation}
    \cW(t)=\int_{-\infty}^t e^{\cL(t-s)}d W(s).
\end{equation}
%
Naturally, $\Exp{}\cW\equiv0$, and we already have $\cC(t,t')$ denoting the covariance of $(\cW(t),\cW(t'))$. Following Gardiner, $\cC(t,t)=\cC(0,0)$ in the stationary limit, is solution of the (well-posed) Sylvester equation\footnote{It is even a Lyapunov equation.}: $\cL\cC(t,t)+\cC(t,t)\cL^{T}=-\cI_N$. Furthermore, if $t>t'$,
$\cC(t,t')=e^{\cL(t-t')}\cC(t,t)$ ($t'<t$ recovered via $\cC(t,t')=\cC(t',t)^T$).

We use these expressions to discretize \eqref{eq:SolSDGen} under the assumption that $F$ is constant over $[t_l,t_{l+1})$\footnote{One big advantage is that we do not even need to assume that $F$ is constant, just that the input doesn't change. For instance it would work the same assuming $F$ is a sinusoid.}.
\begin{equation}
    \vX(t_{l+1})=e^{\delta\cL}\vX(t_{l})+\cL^{-1}(e^{\delta\cL}-\cI_N)F(t_l)+\cW(t_{l+1})-\cW(t_{l})
\end{equation}
Letting $\cM=e^{\delta\cL}$, we obtain 
\begin{equation}
\begin{aligned}
    \vX(t_{l})=&\sum_{n=0}^{l-1}\cM^{l-n-1}\cL^{-1}(\cM-\cI_N)F(t_n)
    \\
    &+\sum_{n=-\infty}^{l}\cM^{l-n}\cW(t_{n})-\sum_{n=-\infty}^{l}\cM^{l-n}\cW(t_{n-1})
    \\
    =&(\cM-\cI_N)\sum_{n=0}^{l-1}\cM^{l-n-1}\cL^{-1}F(t_n)
    \\
    &+W(t_l)+(\cM-\cI_N)\sum_{n=-\infty}^{l-1}\cM^{l-n-1}\cW(t_{n}).
\end{aligned}
\end{equation}


\begin{lemma}

    \left\{
    \sum_{n'=-\infty}^{l'-1}\cC(t_l,t_{n'}){(\cM^T)}^{l'-n'-1}
    \right\}(\cM^T-\cI_N)
    \\
    +
    (\cM-\cI_N)\sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{l'})
    \\
    +
    (\cM-\cI_N)
    \left\{
        \sum_{n'=-\infty}^{l'-1}\sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    \right\}
    (\cM^T-\cI_N).
\end{multline}

With $\cC(0,0):=\cC_0$, we have
$$
\cC(t_{n},t_{n'})=
\begin{cases}
\cM^{n-n'}\cC_0& \text{ if } n\geq n',
\\
\cC_0 {(\cM^{T})}^{n'-n}& \text{ if } n< n'.
\end{cases}
$$
Consider
$$
\cC_{1}:=\sum_{n'=-\infty}^{l'-1}\cM^{l'-n'-1}\cC_0{(\cM^T)}^{l'-n-1}=\sum_{k=0}^{\infty}\cM^{k}\cC_0{(\cM^T)}^{k}.
$$
The matrix $\cC_{1}$ satisfies the Sylvester equation $\cC_1-\cM\cC_1\cM^T=\cC_0$. We have
\begin{equation}\label{E:term2}
\sum_{n'=-\infty}^{l'-1}\cC(t_l,t_{n'}){(\cM^T)}^{l'-n'-1}
=
\sum_{n'=-\infty}^{l'-1}\cM^{l-n'}\cC_0{(\cM^T)}^{l'-n'-1}
=
\cM^{l-l'+1}\cC_{1}
,
\end{equation}
and
\begin{equation}\label{E:term3}
\begin{aligned}
\sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{l'})
=
&\sum_{n=-\infty}^{l'-1}{\cM}^{l-n-1}\cC_0 {(\cM^{T})}^{l'-n}
+
\sum_{n=l'}^{l-1}{\cM}^{(l-n-1)+(n-l')}\cC_0
\\
=&
\cM^{l-l'}\cC_{1}\cM^T+(l-l')\cM^{l-l'-1}\cC_{0}.
\end{aligned}
\end{equation}
Finally, since $n'\leq l'-1<l-1$, we can divide the sum
\begin{equation}
\begin{aligned}
\sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    =&\sum_{n=-\infty}^{n'-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    \\
    &+
    \sum_{n=n'}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
\end{aligned}
\end{equation}
so that 
\begin{equation}
\begin{aligned}
    \sum_{n=-\infty}^{n'-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    =&
    \sum_{n=-\infty}^{n'-1}{\cM}^{l-n-1}\cC_0{(\cM^T)}^{n'-n}{(\cM^T)}^{l'-n'-1}
    \\
    =&
    \sum_{n=-\infty}^{n'-1}{\cM}^{l-n-1}\cC_0{(\cM^T)}^{l'-n-1}
    \\
    =&
    {\cM}^{l-n'}\cC_1{(\cM^T)}^{l'-n'}
\end{aligned}
\end{equation}
and
\begin{equation}
\begin{aligned}
    \sum_{n=n'}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    =&
    \sum_{n=n'}^{l-1}{\cM}^{l-n-1}{\cM}^{n-n'}\cC_0{(\cM^T)}^{l'-n'-1}
    \\
    =&
    \sum_{n=n'}^{l-1}{\cM}^{l-n'-1}\cC_0{(\cM^T)}^{l'-n'-1}
    \\
    =&
    (l-n'){\cM}^{l-n'-1}\cC_0{(\cM^T)}^{l'-n'-1}.
\end{aligned}
\end{equation}
Then
\begin{equation}
\begin{aligned}
\sum_{n'=-\infty}^{l'-1}
    \left\{
        \sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    \right\} 
=&
\sum_{n'=-\infty}^{l'-1}{\cM}^{l-n'}\cC_1{(\cM^T)}^{l'-n'}
\\
&+
\sum_{n'=-\infty}^{l'-1}(l-n'){\cM}^{l-n'-1}\cC_0{(\cM^T)}^{l'-n'-1}
\end{aligned}
\end{equation}
Denote
$$
\cC_{2}:=\sum_{n'=-\infty}^{l'-1}\cM^{l'-n'-1}\cC_1{(\cM^T)}^{l'-n'-1}=\sum_{k=0}^{\infty}\cM^{k}\cC_1{(\cM^T)}^{k},
$$
solution of the Sylvester equation $\cC_2-\cM\cC_2\cM^T=\cC_1$,
and 
$$
\cC_{1}':=\sum_{n'=-\infty}^{l'-1}(l'-n'-1){\cM}^{l'-n'-1}\cC_0{(\cM^T)}^{l'-n'-1}=\sum_{k=1}^{\infty}k \cM^{k}\cC_0{(\cM^T)}^{k}.
$$
Some sum manipulation shows that $\cC_1'$ is solution to the Sylvester equation 
$\cC_1'-\cM\cC_1'\cM^T=\cC_1-\cC_0=\cM\cC_1\cM^T$.

Eventually,
\begin{equation}\label{E:term4}
\sum_{n'=-\infty}^{l'-1}
    \left\{
        \sum_{n=-\infty}^{l-1}{\cM}^{l-n-1}\cC(t_n,t_{n'}){(\cM^T)}^{l'-n'-1}
    \right\} 
    =\cM^{l-l'+1} \cC_2 \cM^T+(l-l'+1)\cM^{l-l'}\cC_1+\cM^{l-l'}\cC_1'.
\end{equation}

The conclusion of the computations is
\begin{multline}
    \mathrm{Cov}(\varepsilon(t_l),\varepsilon(t_{l'}))
    =
    \cM^{l-l'}\cC_0+\cM^{l-l'+1}\cC_{1}(\cM^T-\cI_N)
    \\
    +
    (\cM-\cI_N)\cM^{l-l'}\cC_{1}\cM^T+(l-l')(\cM-\cI_N)\cM^{l-l'-1}\cC_{0}
    \\
    +
    (\cM-\cI_N)\cM^{l-l'}
    \left\{
        \cM \cC_2 \cM^T+(l-l'+1)\cC_1+\cC_1'
    \right\}
    (\cM^T-\cI_N).
\end{multline}
This can be slightly simplified. Recall that $\cC_1-\cM\cC_1\cM^T=\cC_0$, $\cC_2-\cM\cC_2\cM^T=\cC_1$ and $\cC_1'-\cM\cC_1'\cM^T=\cC_1-\cC_0$. This yields that $\cC'_1=\cC_2-\cC_1$. Then 
$$

$$

\paragraph{An interesting formula for the derivative of the matrix exponential.}

\begin{lemma}
For any $t>0$, 
\begin{equation}
\partial_{\theta_p} e^{t\cL}
=
\begin{pmatrix}
    \cI_N & 0_N
\end{pmatrix}
\exp\left( t
\begin{pmatrix}
    \cL &\partial_{\theta_p} \cL
    \\
    0&\cL
\end{pmatrix}
\right)
\begin{pmatrix}
    0_N 
    \\
    \cI_N 
\end{pmatrix}
\end{equation}
\end{lemma}
\begin{proof}
Let $\cA=\begin{pmatrix}
    \cL &\partial_{\theta_p} \cL
    \\
    0&\cL
    \end{pmatrix}$.
We prove the identity:
\begin{equation}
    \exp t
    \cA
    =
    \begin{pmatrix}
    e^{t\cL} &\partial_{\theta_p} e^{t\cL}
    \\
    0&e^{t\cL}
    \end{pmatrix}.
\end{equation}
First, we can check that: 
\begin{equation}
    \exp t
    \cA
    =
    \begin{pmatrix}
    e^{t\cL} &F(t)
    \\
    0&e^{t\cL}
    \end{pmatrix}
\end{equation}
with $F(t)=\int_0^t e^{(t-s)\cL}\partial_{\theta_p} \cL e^{s\cL}ds$.
This follows from the fact that $F(0)=0$ and $F'(t)=\cL F(t)+\partial_{\theta_p} \cL e^{t\cL}$.
Then we can recognize that $F(t)=\partial_{\theta_p}e^{t\cL}$, for instance from \cite[Equation (2.1)]{Wilcox1967}.
\end{proof}

\begin{corollary}
Let $\delta>0$, let $\cM=e^{\delta \cL}$. Let 
$$
\bar{\cM}_p=\exp\left( \delta
\begin{pmatrix}
    \cL &\partial_{\theta_p} \cL
    \\
    0&\cL
\end{pmatrix}
\right)
$$
For every positive integer $l$,
$$
\partial_{\theta_p}(\cM^l)
=
\begin{pmatrix}
    \cI_N & 0_N
\end{pmatrix}
{\left(\bar{\cM}_p\right)}^l
\begin{pmatrix}
    0_N 
    \\
    \cI_N 
\end{pmatrix}
$$
\end{corollary}
